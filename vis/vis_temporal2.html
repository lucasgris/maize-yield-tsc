from datetime import datetime

def extract_date(token):
    date = ''
    for s in token:
        if s.isdigit():
            date += s
    return datetime.strptime(date, '%y%m%d')

def show(mat):

    plt.colorbar()

def remove_no_data(mat, default='zeros'):
    mat[mat <= 0] = np.NaN
    if default == 'zeros':
        mat[np.isnan(mat)] = 0
    elif default == 'ones':
        mat[np.isnan(mat)] = 1
    return mat
